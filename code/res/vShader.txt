#version 330
in vec3 in_Position;
in vec3 in_Normal;
in vec2 in_UV;
out vec4 vert_Normal; 
out vec4 vert_wPos;
uniform mat4 mvpMat;
uniform mat4 objMat;
uniform mat4 mv_Mat;
uniform mat4 nv_Mat;
void main() {
vert_Normal =  nv_Mat * vec4(in_Normal, 0.0); 
gl_Position = mvpMat * objMat * vec4(in_Position, 1.0); 
vert_wPos = objMat * vec4(in_Position, 1.0); 
}
